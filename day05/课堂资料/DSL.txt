# 查询  ------- term词条查询 不分词

GET /user/_search
{
  "query": {
    "term": {
      "city": {
        "value": "深圳"
      }
    }
  }
}

# 查询  ------- terms 词条s 查询 不分词

GET /user/_search
{
  "query": {
    "terms": {
      "city": [
        "深圳",
        "武汉"
      ]
    }
  }
}

# 排序

GET /user/_search
{
  "query": {
    "match_all": {}
  }, 
  "sort": [
    {
      "age": {
        "order": "asc"
      }
    }
  ]
}

# 分页 
#  "from": 0, 表示分页开始的位置: (page-1)*rows
# "size": 20 表示每页显示的行

GET /user/_search
{
  "query": {
    "match_all": {}
  }, 
  "sort": [
    {
      "age": {
        "order": "asc"
      }
    }
  ],
  "from": 0,
  "size": 20
}
# 范围查询  
# "gte": 10,  表示 >= 10   gt    >10 
# "lte": 20,    <=20    lt  <20

GET /user/_search
{
  "query": {
    "range": {
      "age": {
        "gte": 18,
        "lte": 20
      }
    }
  }
}

# 查询 存在 字段为address的数据

GET /user/_search
{
  "query": {
    "exists":{
      "field":"address"
    }
  }
}

# bool查询 (多个条件组合查询)
# 查询 年龄在18 -50 的之间   并且
# must (AND ) must_not (NOT) shoud (OR )
# 查询 城市在 深圳

GET /user/_search
{
  "query": {
    "bool": {
      "must": [
        {
          "range": {
            "age": {
              "gte": 0,
              "lte": 20
            }
          }
        },{
          "term": {
            "city": {
              "value": "深圳"
            }
          }
        }
      ]
    }
  }
}

# match 查询 (匹配查询) 先分词 再查询

GET  user/_search
{
  "query": {
    "match": {
      "description": "深圳打工"
    }
  }
}

# 前缀 prefix   
GET  user/_search 
{
  "query": {
    "prefix": {
      "name": {
        "value": "王"
      }
    }
  }
}

{
  "query": {
    "multi_match": {

        "description"
      ]
    }
  }
